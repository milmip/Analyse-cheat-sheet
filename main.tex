Definition : $s_{n} := \sum_{k=0}^{n} a_n$.

Garder en tête:
\[
\sum \frac{1}{n ^{\alpha}} \left\{ 
    \begin{array}{lcl}
        < \infty & \text{si} & \alpha > 1 \\
        = \infty & \text{si} & \alpha \leq 0 
    \end{array}
\right.
.\]

\begin{itemize}
	\item Convergence $\iff $ queue de série $\longrightarrow 0$.
	\item $x_{n} \to  \neq 0  \quad  \implies $ divergence.

\section*{Fonctions}

\textbf{Limite ponctuelle}: $\lim_{x \to a} f(x)$
\[
\forall \epsilon > 0, \exists \delta > 0, x \in D(f), 0 < |x-a| < \delta \implies |f(x) -l| < \epsilon
.\] 

\textbf{Outils}
\begin{itemize}
	\item Lien avec les suites.
	\item Sandwich.
	\item Opérations.
	\item Compositions.
\end{itemize}

\textbf{Continuité} : si $\lim_{x \to a} f(x) = f(a)$

\textbf{Implications} (sur intervals fermés)
\begin{itemize}
	\item Min-max.
	\item TVI.
\end{itemize}

\textbf{Continuité uniforme}
\[
\forall \epsilon >0, \exists \delta > 0, x,y \in D(f). |x-y| < \delta \implies |f(x) - f(y)| < \epsilon
.\] 

\textbf{Outils}

(pour prouver)
\begin{itemize}
	\item Definition.
	\item Intervals fermés, ok.
	\item Intervals ouverts, ok \emph{si prolongeable} (même à l'infini).
	\item $f$ est $M$-lipschitzienne. (suffisant mais pas nécessaire)
\end{itemize}
(pour réfuter)
\begin{itemize}
	\item Suites $x_{n}, y_{n}$ particulières.
\end{itemize}

\section*{Dérivées}
\textbf{Rolle} : ($f$ continue dérivable) $\exists c \text{ t.q. } f'(c) = 0$.

\textbf{TAFG} : (=) $\exists c \text{ t.q. } \left[ f(b) - f(a) \right] g'(c) = \left[ g(b) - g(a) \right] f'(c)$.

\textbf{TAF} : (=) $\exists c \text{ t.q. } ( f(b) - f(a) ) = ( b - a ) f'(c)$.

\textbf{Croissance}: $f \nearrow \iff f'(x) \bf{\geq } 0$, ect.

\textbf{$L$-Lipschitzienne}: $\exists L\in \R_{+}$ t.q. 
\[
\forall x, y \in D(f), |f(x) - f(y)| \leq L |x - y|
.\] 

\textbf{Extremas} : \textbf{1}. $\in \{a, b\} $, \textbf{2}. pt. crit., \textbf{3} . pt. $f$ non dérivable.

\textbf{Convexité} : $f$ est convexe sur $(a, b)$ si :
\[
\forall \lambda \in  [0, 1], \forall x_1, x_2 \in (a, b), f(\lambda x_1 + (1-\lambda)x_2) \leq \lambda f(x_1) + (1-\lambda)f(x_2)
.\] 
\textbf{Concavité} : $-f$ convexe.

\textbf{Implications} 
\begin{itemize}
	\item $f$ continue !
	\item $\iff f' \nearrow $.
	\item $\iff f'' \geq 0 $.
\end{itemize}

\section*{Suites de fonctions}
\textbf{Convergence ponctuelle} 
\[
	\forall \epsilon > 0, \forall x \in E, \exists N(\epsilon, x) \in \N, \forall  n \geq N , |f_{n}(x) - f(x)| < \epsilon
.\] 
\textbf{Convergence uniforme} 
\[
	\forall \epsilon > 0, \exists N(\epsilon) \in \N, \forall  n \geq N , |f_{n}(x) - f(x)| < \epsilon, \forall x \in E
.\] 
\textbf{Uniformité de Cauchy}
\[
	\forall \epsilon > 0, \exists N(\epsilon) \in \N, \forall  n,m \geq N , |f_{n}(x) - f_{m}(x)| < \epsilon, \forall x \in E
.\] 

\textbf{Outils} 
\begin{itemize}
	\item Non conv. ponct. $\implies$ Non conv. unif.
	\item (prouver conv. u.) $\lim_{n \to \infty} \sup \{|f_{n}(x) - f(x)|,\quad  x \in E\} $.
	\item Unif. de Cauchy $\iff $ unif. convergence.
	\item Combinaison \textbf{préserve} la conv. unif.
	\item \emph{Dini} sur les suites monotones.
\end{itemize}
\textbf{Implications} : si $f_{n} \xlongrightarrow{U} f$
\begin{itemize}
	\item Alors $\lim_{x \to a} \lim_{n \to \infty} f_{n}(x) = \lim_{n \to \infty} \lim_{x \to a} f_{n}(x)$ 
	\item Si $f_{n} \in \cC^{0}$, alors  $f \in \cC^{0} $ (aussi pour réfuter, si $f \not\in \cC^{0}$).
\end{itemize}

\section*{Séries entières}
Définition : $f(x) =  \sum_{n=0}^{\infty} a_n x^n$.

Domaine : $D(f) := \{x \text{ t.q. } f(x) < \infty\}  = [-R, R)$ (ouvert/fermé à voir les cas).

Rayon : $R := \sup |D(f)|$.

\textbf{Implications}:  $\forall 0 < \rho < R$
\[
	\sum_{k=0}^{n} |a_k x^k| \xlongrightarrow{U} f(x), \text{ sur } [-\rho, \rho] \text{ (puissant)}
.\] 

\textbf{Outils} (pour trouver le rayon)
\begin{itemize}
	\item Définition.
	\item $R = 1/\lim_{n \to \infty} \sup \sqrt[n]{|a_{n}|}$.
	\item $R = 1/\lim_{n \to \infty} \l|\frac{a_{n+1}}{a_{n}}\r| $
\end{itemize}
\textbf{Implications} 
\begin{itemize}
	\item $f \in \cC^{\infty}((-R, R), \R) $.
\end{itemize}
